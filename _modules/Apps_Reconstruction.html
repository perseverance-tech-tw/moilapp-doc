class Reconstruction:
    """
    Reconstruction class manages image reconstruction operations in a PyQt application.

    Attributes:
        parent: Reference to the MainWindow of the application.
        show: Instance of ShowImageResult to display images.

    Methods:
        - __init__: Initializes the Reconstruction object.
        - reconstructionImage: Performs image reconstruction and displays the result.
    """
    def __init__(self, MainWindow):
        self.parent = MainWindow
        self.show = ShowImageResult(self.parent)

    def reconstructionImage(self):
        """
        Performs image reconstruction and displays the result.

        If an image is available, it displays the original image and any associated annotations.
        """
        if self.parent.image is None:
            pass
        else:
            self.show.showOriginalImage(self.parent.image)
            self.show.show_development()
